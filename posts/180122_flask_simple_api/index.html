from logging import FileHandler, INFO, Formatter


app = Flask(__name__)

# log
err_log = FileHandler('error_log_'+time.strftime('%Y%m%d',time.localtime()))

err_log.setFormatter(logging_format)
app.logger.addHandler(err_log)
